1//! # Parameter Provider Controller
2//!
3//! This module defines the data structures and controller logic for managing
4//! "Parameter Providers" within the system. It includes:
5//!
6//! * The main `Controller` struct for interacting with the API.
7//! * Data Transfer Objects (Dtos) like `ParameterProviderEntity` and `ParameterProviderDto`
8//!     which map directly to the JSON aPI.
9//! * Various supporting enums and Dtos that describe properties, allowable values,
10//!     and component status.
11//!
12//! The primary entities are serialized and deserialized with `serde` using `camelCase`
13//! conventions to match the target JSON API.
14
15use crate::api::{ParameterProviderEntity};
16use crate::common::client::HttpClient;
17use crate::common::config::Config;
18use serde::{Deserialize, Serialize};
19use std::sync::Arc;
20
21/// Manages operations related to Parameter Providers.
22///
23/// This controller holds a shared `HttpClient` and `Config` to make
24/// API requests to the controller endpoints.
25#[derive(Debug)]
26pub struct Controller {
27    /// A thread-safe, shared HTTP client for making API requests.
28    client: Arc<HttpClient>,
29    /// A thread-safe, shared configuration object, primarily for the API base URL.
30    config: Arc<Config>,
31}
32
33impl Controller {
34    /// Creates a new `Controller` instance.
35    ///
36    /// # Arguments
37    ///
38    /// * `client` - An `Arc<HttpClient>` used to make API requests.
39    /// * `config` - An `Arc<Config>` containing the API base URL.
40    pub fn new(client: Arc<HttpClient>, config: Arc<Config>) -> Self {
41        Self { client, config }
42    }
43
44    /// Creates a new Parameter Provider on the controller.
45    ///
46    /// This function performs an HTTP POST to the `/controller/parameter-providers`
47    /// endpoint.
48    ///
49    /// # Arguments
50    ///
51    /// * `payload` - A `ParameterProviderEntity` describing the new provider.
52    ///               The `Default::default()` implementation is a good
53    ///               starting point.
54    ///
55    /// # Errors
56    ///
57    /// Returns an `anyhow::Error` if the HTTP request fails, if the API
58    /// returns an error status code, or if the response cannot be
59    /// deserialized into a `ParameterProviderEntity`.
60    pub async fn post_parameter_providers(
61        &self,
62        payload: &ParameterProviderEntity,
63    ) -> anyhow::Result<ParameterProviderEntity> {
64        let response = self
65            .client
66            .post_json::<ParameterProviderEntity, ParameterProviderEntity>(
67                &format!(
68                    "{}/controller/parameter-providers",
69                    self.config.api_base_url
70                ),
71                payload,
72            )
73            .await?;
74        Ok(response)
75    }
76}
