1//! # Parameter Provider Controller
2//!
3//! This module defines the data structures and controller logic for managing
4//! "Parameter Providers" within the system. It includes:
5//!
6//! * The main `Controller` struct for interacting with the API.
7//! * Data Transfer Objects (DTOs) like `ParameterProviderEntity` and `ParameterProviderDTO`
8//!     which map directly to the JSON aPI.
9//! * Various supporting enums and DTOs that describe properties, allowable values,
10//!     and component status.
11//!
12//! The primary entities are serialized and deserialized with `serde` using `camelCase`
13//! conventions to match the target JSON API.
14

24/// Manages operations related to Parameter Providers.
25///
26/// This controller holds a shared `HttpClient` and `Config` to make
27/// API requests to the controller endpoints.
28#[derive(Debug)]
29pub struct Controller {
30    /// A thread-safe, shared HTTP client for making API requests.
31    client: Arc<HttpClient>,
32    /// A thread-safe, shared configuration object, primarily for the API base URL.
33    config: Arc<Config>,
34}

36/// Represents a Parameter Provider entity, which includes the component itself
37/// along with its metadata, permissions, and position on the canvas.
38#[derive(Debug, Deserialize, Serialize)]
39#[serde(rename_all = "camelCase")]
40pub struct ParameterProviderEntity {
41    /// Any bulletins (warnings, errors) associated with this parameter provider.
42    pub bulletins: Option<Vec<BulletinEntity>>,
43    /// The core DTO for the parameter provider component.
44    pub component: Option<ParameterProviderDTO>,
45    /// The unique identifier for the parameter provider.
46    pub id: Option<String>,
47    /// Permissions for accessing and modifying this entity.
48    pub permissions: Option<PermissionsDTO>,
49    /// The position of the component on the UI canvas.
50    pub position: Option<PositionDTO>,
51    /// The revision information for optimistic locking.
52    pub revision: Option<RevisionDTO>,
53    /// The URI for this entity, for direct API access.
54    pub uri: Option<String>,
55}

57/// The core Data Transfer Object for a Parameter Provider.
58///
59/// This structure contains the configuration, properties, and status
60/// of a specific parameter provider component.
61#[derive(Debug, Deserialize, Serialize)]
62#[serde(rename_all = "camelCase")]
63pub struct ParameterProviderDTO {
64    /// Components that are affected by this parameter provider.
65    pub affected_components: Option<Vec<AffectedComponentEntity>>,
66    /// Any annotation data associated with the component.
67    pub annotation_data: Option<String>,
68    /// The bundle (group, artifact, version) that defines this component's type.
69    pub bundle: Option<BundleDTO>,
70    /// User-supplied comments.
71    pub comments: Option<String>,
72    /// An optional URL for a custom configuration UI.
73    pub custom_ui_url: Option<String>,
74    /// Whether this component is deprecated.
75    pub deprecated: Option<bool>,
76    /// Descriptions of the properties this component supports.
77    pub descriptors: Option<HashMap<String, PropertyDescriptorDTO>>,
78    /// Whether the extension (NAR) for this component is missing.
79    pub extension_missing: Option<bool>,
80    /// The unique identifier for this component.
81    pub id: Option<String>,
82    /// Whether multiple versions of this component's bundle are available.
83    pub multiple_versions_available: Option<bool>,
84    /// The user-defined name of the component.
85    pub name: Option<String>,
86    /// Configuration for parameter groups, mapping groups to parameter contexts.
87    pub parameter_group_configurations: Option<Vec<ParameterGroupConfigurationEntity>>,
88    /// The status of parameters provided by this component.
89    pub parameter_status: Option<ParameterStatusDTO>,
90    /// The ID of the parent process group.
91    pub parent_group_id: Option<String>,
92    /// Whether this component persists its state.
93    pub persists_state: Option<bool>,
94    /// The component's position on the canvas.
95    pub position: Option<PositionDTO>,
96    /// The configured properties (key-value pairs) for this component.
97    pub properties: Option<HashMap<String, Option<String>>>,
98    /// Parameter contexts that reference this provider.
99    pub referencing_parameter_contexts: Option<Vec<ParameterProviderReferencingComponentEntity>>,
100    /// Whether this component has restricted functionality.
101    pub restricted: Option<bool>,
102    /// The fully-qualified class name of the component's implementation.
103    #[serde(rename = "type")]
104    pub _type: Option<String>,
105    /// Any validation errors associated with the component's configuration.
106    pub validation_errors: Option<Vec<String>>,
107    /// The current validation status (e.g., VALID, INVALID).
108    pub validation_status: Option<ValidationStatus>,
109    /// The ID of this component in a version-controlled flow.
110    pub versioned_component_id: Option<String>,
111}

148/// The core Data Transfer Object for a Parameter.
149#[derive(Debug, Deserialize, Serialize)]
150#[serde(rename_all = "camelCase")]
151pub struct ParameterDTO {

153    pub description: Option<String>,

157    pub name: Option<String>,
158    /// A reference to the parameter context this parameter belongs to.
159    pub parameter_context: Option<ParameterContextReferenceEntity>,
160    /// Whether this parameter is provided (e.g., by a provider) vs. user-entered.
161    pub provided: Option<bool>,
162    /// A list of assets (files) referenced by this parameter.
163    pub referenced_assets: Option<Vec<AssetReferenceDTO>>,

166    /// Whether this parameter is sensitive (its value should be hidden).
167    pub sensitive: Option<bool>,
168    /// The value of the parameter. This may be `None` if sensitive.
169    pub value: Option<String>,
170    /// Whether the value was recently removed.
171    pub value_removed: Option<bool>
172}

174/// An enum representing the status of a parameter in relation to a
175/// parameter context.
176#[derive(Debug, Deserialize, Serialize)]
177pub enum StatusType {
178    /// The parameter is new and does not exist in the context.
179    #[serde(rename="NEW")]
180    New,
181    /// The parameter exists in the context but its value has changed.
182    #[serde(rename="CHANGED")]
183    Changed,
184    /// The parameter has been removed.
185    #[serde(rename="REMOVED")]
186    Removed,
187    /// The parameter is missing but is still referenced.
188    #[serde(rename="MISSING_BUT_REFERENCED")]
189    MissingButReferenced,
190    /// The parameter exists and its value is unchanged.
191    #[serde(rename="UNCHANGED")]
192    Unchanged,
193}

366impl Controller {
367    /// Creates a new `Controller` instance.
368    ///
369    /// # Arguments
370    ///
371    /// * `client` - An `Arc<HttpClient>` used to make API requests.
372    /// * `config` - An `Arc<Config>` containing the API base URL.
373    pub fn new(client: Arc<HttpClient>, config: Arc<Config>) -> Self {
374        Self { client, config }
375    }
376
377    /// Creates a new Parameter Provider on the controller.
378    ///
379    /// This function performs an HTTP POST to the `/controller/parameter-providers`
380    /// endpoint.
381    ///
382    /// # Arguments
383    ///
384    /// * `payload` - A `ParameterProviderEntity` describing the new provider.
385    ///               The `Default::default()` implementation is a good
386    ///               starting point.
387    ///
388    /// # Errors
389    ///
390    /// Returns an `anyhow::Error` if the HTTP request fails, if the API
391    /// returns an error status code, or if the response cannot be
392    /// deserialized into a `ParameterProviderEntity`.
393    pub async fn post_parameter_providers(
394        &self,
395        payload: &ParameterProviderEntity,
396    ) -> anyhow::Result<ParameterProviderEntity> {
397        let response = self
398            .client
399            .post_json::<ParameterProviderEntity, ParameterProviderEntity>(
400                &format!("{}/controller/parameter-providers", self.config.api_base_url),
401                payload,
402            )
403            .await?;
404        Ok(response)
405    }
406}
