1//! # Access Module
2//!
3//! Provides high-level bindings for the NiFi "access" API.
4//!
5//! This module encapsulates the logic for authenticating (getting a token)
6//! and logging out (invalidating and clearing the token).
7
8// Note: These `use` statements are assumed to be correct based on your project's structure.
9use crate::common::client::HttpClient;
10use crate::common::config::Config;
11use serde_json::json;
12use std::sync::Arc;
13
14/// A service for interacting with NiFi's access and authentication endpoints.
15///
16/// It is instantiated with shared (`Arc`) instances of `HttpClient` and `Config`.
17/// Actions performed here (like `get_access_token`) will affect the state
18/// of the shared `HttpClient`.
19pub struct Access {
20    client: Arc<HttpClient>,
21    config: Arc<Config>,
22}
23
24impl Access {
25    /// Creates a new instance of the `Access` service.
26    ///
27    /// # Arguments
28    ///
29    /// * `client` - The shared `HttpClient` to be used for requests.
30    /// * `config` - The application configuration (containing `api_base_url`, `username`, etc.).
31    pub fn new(client: Arc<HttpClient>, config: Arc<Config>) -> Self {
32        Access { client, config }
33    }
34
35    /// Attempts to authenticate against the NiFi API using credentials from `Config`.
36    ///
37    /// Sends a `POST` to `/access/token` with the username and password.
38    ///
39    /// On success, it **atomically updates the shared `HttpClient`** with the new
40    /// token, so all future API requests will use it.
41    ///
42    /// # Errors
43    ///
44    /// Returns `HttpClientError` if the request fails (e.g., `HttpError` 401
45    /// for bad credentials, or `RequestError` if the server is unreachable).
46    pub async fn get_access_token(&self) -> anyhow::Result<String> {
47        let response = self
48            .client
49            .post_form::<_, String>(
50                &format!("{}/access/token", self.config.api_base_url),
51                &json!({
52                    "username": self.config.username,
53                    "password": self.config.password,
54                }),
55            )
56            .await?;
57
58        // Store the token in the shared client
59        self.client.set_auth_token(response.clone()).await?;
60
61        Ok(response)
62    }
63
64    /// Logs out of the NiFi API.
65    ///
66    /// Sends a `DELETE` request to `/access/logout`.
67    ///
68    /// On success, it **atomically clears the token from the shared `HttpClient`**,
69    /// effectively logging the client out.
70    ///
71    /// # Errors
72    ///
73    /// Returns `HttpClientError` if the `DELETE` request fails.
74    pub async fn logout(&self) -> anyhow::Result<()> {
75        // Call the logout endpoint. We expect an empty '()' response.
76        let response = self
77            .client
78            .delete::<()>(&format!("{}/access/logout", self.config.api_base_url))
79            .await?;
80
81        // Clear the token from the shared client
82        self.client.clear_auth_token().await?;
83
84        Ok(response)
85    }
86}

88/// Integration tests for the Access module.
89///
90/// These tests perform real network calls to a local NiFi endpoint.
91/// (Ensure NiFi is running locally and `Config::default()`
92/// has the correct credentials).
93#[cfg(test)]
94mod test {
95    use super::*;
96    use tracing_test::traced_test;
97
98    // Note: `use crate::common::config::Config` might be needed here
99    // if `Config::default()` isn't in scope.
100
101    #[tokio::test]
102    #[traced_test]
103    async fn test_get_access_token() {
104        let client = Arc::new(HttpClient::new());
105        let config = Arc::new(Config::default()); // Assumes correct credentials
106        let access = Access::new(client.clone(), config.clone());
107
108        let access_token = access.get_access_token().await;
109
110        tracing::info!("{:#?}", access_token);
111        assert!(
112            access_token.is_ok(),
113            "test_get_access_token call error: {:?}",
114            access_token
115        );
116    }
117
118    #[tokio::test]
119    #[traced_test]
120    async fn test_get_access_token_fail() {
121        let client = Arc::new(HttpClient::new());
122        let config = Arc::new(Config {
123            password: "false_password".to_string(),
124            ..Default::default()
125        });
126        let access = Access::new(client.clone(), config.clone());
127
128        let access_token = access.get_access_token().await;
129
130        tracing::info!("{:#?}", access_token);
131        // We expect this to fail (is_err())
132        assert!(
133            access_token.is_err(),
134            "test_get_access_token should have failed, but returned Ok: {:?}",
135            access_token
136        );
137    }
138
139    #[tokio::test]
140    #[traced_test]
141    async fn test_logout() {
142        // --- 1. Setup ---
143        let config = Arc::new(Config::default());
144        let client = Arc::new(HttpClient::new());
145        let access = Access::new(client.clone(), config.clone());
146
147        // --- 2. Check initial state (no token) ---
148        let initial_token_result = client.get_auth_token().await;
149        assert!(initial_token_result.is_ok(), "get_auth_token failed: {:?}", initial_token_result);
150        assert!(initial_token_result.unwrap().is_none(), "Token should be None at start");
151        tracing::info!("Initial state: Logged-out (OK)");
152
153        // --- 3. Log in ---
154        let login_result = access.get_access_token().await;
155        assert!(login_result.is_ok(), "Login failed: {:?}", login_result);
156        let token_str = login_result.unwrap();
157        tracing::info!("Login successful, token [REDACTED]");
158
159        // --- 4. Check post-login state (has token) ---
160        let token_after_login_result = client.get_auth_token().await;
161        assert!(token_after_login_result.is_ok(), "get_auth_token failed: {:?}", token_after_login_result);
162        let token_after_login = token_after_login_result.unwrap();
163        assert!(token_after_login.is_some(), "Token should be Some after login");
164        assert_eq!(token_after_login.unwrap(), token_str, "Stored token does not match");
165
166        // --- 5. Log out ---
167        let logout_result = access.logout().await;
168        assert!(logout_result.is_ok(), "Logout failed: {:?}", logout_result);
169        tracing::info!("Logout successful");
170
171        // --- 6. Check final state (no token) ---
172        let final_token_result = client.get_auth_token().await;
173        assert!(final_token_result.is_ok(), "get_auth_token failed: {:?}", final_token_result);
174        assert!(final_token_result.unwrap().is_none(), "Token should be None after logout");
175        tracing::info!("Final state: Logged-out (OK)");
176    }
177}
