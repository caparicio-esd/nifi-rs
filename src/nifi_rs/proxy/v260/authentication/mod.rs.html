1//! # Authentication Module
2//!
3//! Provides bindings for the NiFi "authentication" API endpoints.
4//!
5//! This module is primarily used to query the server's authentication
6//! configuration, such as whether login is supported or if an external
7//! login flow is required.
8
9use crate::common::client::HttpClient;
10use crate::common::config::Config;
11use crate::proxy::v260::api::AuthenticationConfigurationEntity;
12use std::sync::Arc;
13
14/// A service for interacting with NiFi's authentication configuration endpoints.
15///
16/// It is instantiated with shared (`Arc`) instances of `HttpClient` and `Config`.
17pub struct Authentication {
18    client: Arc<HttpClient>,
19    config: Arc<Config>,
20}
21
22impl Authentication {
23    /// Creates a new instance of the `Authentication` service.
24    ///
25    /// # Arguments
26    ///
27    /// * `client` - The shared `HttpClient` to be used for requests.
28    /// * `config` - The application configuration (containing `api_base_url`).
29    pub fn new(client: Arc<HttpClient>, config: Arc<Config>) -> Self {
30        Self { client, config }
31    }
32
33    /// Fetches the authentication configuration from the NiFi instance.
34    ///
35    /// This method calls the `GET /authentication/configuration` endpoint.
36    ///
37    /// # Errors
38    ///
39    /// Returns `HttpClientError` if the request fails (e.g., network error,
40    /// HTTP status error, or a JSON parsing error).
41    pub async fn get_authentication_configuration(
42        &self,
43    ) -> anyhow::Result<AuthenticationConfigurationEntity> {
44        let response = self
45            .client
46            .get_json::<AuthenticationConfigurationEntity>(&format!(
47                "{}/authentication/configuration",
48                self.config.api_base_url
49            ))
50            .await?;
51        Ok(response)
52    }
53}

65        // --- 1. Setup ---
66        let client = Arc::new(HttpClient::new());
67        let config = Arc::new(Config::default()); // Assumes correct credentials
68        let access = Access::new(client.clone(), config.clone());
69        let _ = access.get_access_token().await;
70
71        // --- 2. Check Configuration Connection ---
72        let authentication = Authentication::new(client.clone(), config.clone());
73        let auth_configuration = authentication.get_authentication_configuration().await;
74        assert!(
75            auth_configuration.is_ok(),
76            "test_get_authentication_configuration call error: {:?}",
77            auth_configuration
78        );
79
80        // --- 3. Assert over object ---
81        let auth_configuration = auth_configuration.unwrap();
82        assert!(auth_configuration.authentication_configuration.is_some(),);
83        debug!("{:#?}", auth_configuration);
84    }
