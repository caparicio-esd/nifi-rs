1//! # Common HTTP Client Module
2//!
3//! Provides a robust, cloneable, and thread-safe `HttpClient`
4//! that internally manages authentication state.
5
6use async_trait::async_trait;
7use serde::Serialize;
8use serde::de::DeserializeOwned;
9use std::sync::Arc;
10use std::time::Duration;
11use thiserror::Error;
12use tokio::sync::RwLock;
13
14/// A cloneable, async, and state-aware HTTP client for making API requests.
15///
16/// This client wraps a `reqwest::Client` and is designed to be safely shared
17/// across multiple threads and services (e.g., wrapped in an `Arc`).
18///
19/// It automatically handles connection pooling, timeouts, and the user-agent.
20///
21/// Its main feature is the management of an **internal, mutable authentication token**.
22/// It allows services like an `Access` module to log in (`set_auth_token`), and then
23/// all subsequent requests from *any* service sharing this client
24/// will automatically include the token.
25///
26/// It's cheap to clone (`#[derive(Clone)]`) because the internal `reqwest::Client`
27/// and the `auth_token` (`Arc<RwLock<...>>`) both use atomic reference counting.
28#[derive(Clone, Debug)]
29pub struct HttpClient {
30    client: reqwest::Client,
31    /// The shared, mutable authentication token.
32    /// `Arc` makes it shareable, `RwLock` makes it safely mutable.
33    /// `Option<String>` represents the state: "logged-in" (`Some(token)`) or "logged-out" (`None`).
34    auth_token: Arc<RwLock<Option<String>>>,
35}
36
37/// Represents all possible errors that can occur during an HTTP request.
38#[derive(Debug, Error)]
39pub enum HttpClientError {
40    /// A network or request-building error from `reqwest`.
41    #[error("HttpClientError::RequestError - {0}")]
42    RequestError(reqwest::Error),
43
44    /// An HTTP status error (4xx or 5xx) returned by the server.
45    #[error("HttpClientError::HttpError - {status}:{message}")]
46    HttpError {
47        status: reqwest::StatusCode,
48        message: String,
49    },
50
51    /// An error during the deserialization (parsing) of the response body.
52    /// Error al leer el cuerpo de la respuesta (ej. fallo de red a mitad).
53    #[error("HttpClientError::BodyReadError - Failed to read response body: {0}")]
54    BodyReadError(reqwest::Error), // Reemplaza el 'ParseError' original
55
56    /// Error al deserializar el cuerpo de la respuesta (ej. no coincide el schema).
57    /// Â¡Este es el error que usted desea!
58    #[error("HttpClientError::DeserializeError - {source}\n--- Raw Text ---\n{raw_text}\n---")]
59    DeserializeError {
60        #[source]
61        source: serde_json::Error,
62        raw_text: String,
63    },
64}
65
66/// Allows for automatic conversion from `reqwest::Error` to `HttpClientError` (using `?`).
67impl From<reqwest::Error> for HttpClientError {
68    fn from(err: reqwest::Error) -> Self {
69        HttpClientError::RequestError(err)
70    }
71}

73/// A newtype wrapper to indicate that a response should be deserialized as JSON.
74///
75/// This is used in generic trait bounds to disambiguate, for example:
76/// `client.post_form::<_, JsonResponse<MyStruct>>(...).await`
77#[derive(Debug)]
78pub struct JsonResponse<T>(pub T);
79
80/// A trait that defines how to parse a `reqwest::Response` into a specific output type.
81///
82/// This allows our generic methods (`post_form`, `delete`) to return
83/// a `String`, a JSON struct (via `JsonResponse`), or nothing (`()`).
84#[async_trait]
85pub trait ApiResponse: Sized {
86    /// Parses the entire response into `Self`.
87    ///
88    /// # Errors
89    /// Returns `HttpClientError::ParseError` if deserialization fails.
90    async fn from_response(response: reqwest::Response) -> anyhow::Result<Self, HttpClientError>;
91}
92
93/// `ApiResponse` implementation for `()`, for when we don't care about the response body.
94#[async_trait]
95impl ApiResponse for () {
96    async fn from_response(_response: reqwest::Response) -> anyhow::Result<Self, HttpClientError> {
97        Ok(()) // Just success, we don't read the body
98    }
99}
100
101/// `ApiResponse` implementation for `String`, to get the body as plain text.
102#[async_trait]
103impl ApiResponse for String {
104    async fn from_response(response: reqwest::Response) -> anyhow::Result<Self, HttpClientError> {
105        response
106            .text()
107            .await
108            .map_err(HttpClientError::BodyReadError)
109    }
110}
111
112/// `ApiResponse` implementation for `JsonResponse<T>`, for deserializing JSON.
113#[async_trait]
114impl<T> ApiResponse for JsonResponse<T>
115where
116    T: DeserializeOwned + Send,
117{
118    async fn from_response(response: reqwest::Response) -> Result<Self, HttpClientError> {
119        let json = response
120            .json::<T>()
121            .await
122            .map_err(HttpClientError::BodyReadError)?;
123        Ok(JsonResponse(json))
124    }
125}

127impl HttpClient {
128    /// Creates a new `HttpClient` with default settings.
129    ///
130    /// The client is initialized without an authentication token (in a "logged-out" state).
131    ///
132    /// # Panics
133    /// Panics if the `reqwest::Client` builder fails (e.g., if the
134    /// system's TLS backend cannot be initialized).
135    pub fn new() -> Self {
136        let client = reqwest::Client::builder()
137            .connect_timeout(Duration::from_secs(5))
138            .timeout(Duration::from_secs(30))
139            .pool_idle_timeout(Duration::from_secs(30))
140            .user_agent(concat!(
141                env!("CARGO_PKG_NAME"),
142                "/",
143                env!("CARGO_PKG_VERSION")
144            ))
145            // Allows self-signed certificates (e.g., from a local NiFi instance)
146            // WARNING: Do not use in production unless strictly necessary.
147            .danger_accept_invalid_certs(true)
148            .build()
149            .expect("Failed to build reqwest client");
150
151        Self {
152            client,
153            // Initialize the token as `None` (logged-out)
154            auth_token: Arc::new(RwLock::new(None)),
155        }
156    }
157
158    /// Safely sets (or overwrites) the internal authentication token.
159    ///
160    /// Acquires a *write* lock on the token.
161    pub async fn set_auth_token(&self, token: String) -> anyhow::Result<()> {
162        let mut guard = self.auth_token.write().await;
163        *guard = Some(token);
164        Ok(())
165    }
166
167    /// Safely clears the internal authentication token (for logout).
168    ///
169    /// Acquires a *write* lock on the token.
170    pub async fn clear_auth_token(&self) -> anyhow::Result<()> {
171        let mut guard = self.auth_token.write().await;
172        *guard = None;
173        Ok(())
174    }
175
176    /// Gets a clone of the current authentication token, if one exists.
177    ///
178    /// Acquires a (cheap) *read* lock on the token.
179    pub async fn get_auth_token(&self) -> anyhow::Result<Option<String>> {
180        let guard = self.auth_token.read().await;
181        Ok(guard.clone())
182    }
183
184    /// Private helper to execute a request, adding authentication and handling errors.
185    ///
186    /// 1. Acquires a *read* lock on the token and adds the `Bearer` header if it exists.
187    /// 2. Sends the request.
188    /// 3. Checks for a successful HTTP status (`error_for_status`), converting 4xx/5xx
189    ///    into `HttpClientError::HttpError`.
190    async fn execute_request(
191        &self,
192        builder: reqwest::RequestBuilder,
193    ) -> anyhow::Result<reqwest::Response, HttpClientError> {
194        // Acquire a read lock
195        let token_guard = self.auth_token.read().await;
196
197        // Add the token ONLY if it exists
198        let builder = if let Some(token) = token_guard.as_ref() {
199            builder.bearer_auth(token)
200        } else {
201            builder
202        };
203
204        let response = builder.send().await?;
205        let response = response
206            .error_for_status()
207            .map_err(|err| HttpClientError::HttpError {
208                status: err.status().unwrap_or(reqwest::StatusCode::BAD_REQUEST),
209                message: err.to_string(),
210            })?;
211        Ok(response)
212    }
213
214    async fn deserialize_json_response<R>(response: reqwest::Response) -> Result<R, HttpClientError>
215    where
216        R: DeserializeOwned,
217    {
218        // 1. Leer el cuerpo a texto
219        let raw_text = response
220            .text()
221            .await
222            .map_err(HttpClientError::BodyReadError)?;
223
224        // 2. Intentar deserializar
225        let result = serde_json::from_str::<R>(&raw_text);
226
227        // 3. Devolver Ok o el nuevo error detallado
228        match result {
229            Ok(data) => Ok(data),
230            Err(serde_error) => Err(HttpClientError::DeserializeError {
231                source: serde_error,
232                raw_text,
233            }),
234        }
235    }
236
237    /// Performs a `GET` request and deserializes the response as JSON.
238    ///
239    /// `R` is the response type (must be `DeserializeOwned`).
240    ///
241    /// # Errors
242    /// Returns `HttpClientError` on network, HTTP, or parsing failure.
243    pub async fn get_json<R>(&self, url: &str) -> anyhow::Result<R, HttpClientError>
244    where
245        R: DeserializeOwned,
246    {
247        let builder = self.client.get(url);
248        let response = self.execute_request(builder).await?;
249        Self::deserialize_json_response(response).await
250    }
251
252    /// Performs a `POST` request with a JSON payload and deserializes the response as JSON.
253    ///
254    /// `T` is the payload type (must be `Serialize`).
255    /// `R` is the response type (must be `DeserializeOwned`).
256    ///
257    /// # Errors
258    /// Returns `HttpClientError` on network, HTTP, or parsing failure.
259    pub async fn post_json<T, R>(
260        &self,
261        url: &str,
262        payload: &T,
263    ) -> anyhow::Result<R, HttpClientError>
264    where
265        T: Serialize,
266        R: DeserializeOwned,
267    {
268        let builder = self.client.post(url).json(payload);
269        let response = self.execute_request(builder).await?;
270        Self::deserialize_json_response(response).await
271    }
272
273    /// Performs a `PUT` request with a JSON payload and deserializes the response as JSON.
274    ///
275    /// (Identical to `post_json`, but uses `PUT`).
276    ///
277    /// # Errors
278    /// Returns `HttpClientError` on network, HTTP, or parsing failure.
279    pub async fn put_json<T, R>(&self, url: &str, payload: &T) -> anyhow::Result<R, HttpClientError>
280    where
281        T: Serialize,
282        R: DeserializeOwned,
283    {
284        let builder = self.client.put(url).json(payload);
285        let response = self.execute_request(builder).await?;
286        Self::deserialize_json_response(response).await
287    }

289    /// Performs a `DELETE` request and parses the response using `ApiResponse`.
290    ///
291    /// `R` is the response type (must implement `ApiResponse`, e.g., `()`, `String`).
292    ///
293    /// # Errors
294    /// Returns `HttpClientError` on network, HTTP, or parsing failure.
295    pub async fn delete<R>(&self, url: &str) -> anyhow::Result<R, HttpClientError>
296    where
297        R: ApiResponse,
298    {
299        let builder = self.client.delete(url);
300        let response = self.execute_request(builder).await?;
301        R::from_response(response).await
302    }

304    /// Performs a `POST` request with a form payload (`x-www-form-urlencoded`).
305    ///
306    /// `T` is the payload type (must be `Serialize`).
307    /// `R` is the response type (must implement `ApiResponse`, e.g., `()`, `String`).
308    ///
309    /// # Errors
310    /// Returns `HttpClientError` on network, HTTP, or parsing failure.
311    ///
312    /// # Example
313    /// ```no_run
314    /// # use serde::{Serialize, Deserialize};
315    /// # use nifi_rs::common::client::{HttpClient, JsonResponse, ApiResponse};
316    /// #
317    /// #[derive(Serialize)]
318    /// struct MyForm {
319    ///     username: String,
320    /// }
321    ///
322    /// #[derive(Deserialize)]
323    /// struct MyResponse {
324    ///     token: String,
325    /// }
326    ///
327    /// # async fn run() -> anyhow::Result<()> {
328    /// # let client = HttpClient::new();
329    /// let form = MyForm { username: "admin".to_string() };
330    ///
331    /// // Request a response as a String (the token)
332    /// let token_str = client.post_form::<_, String>("url", &form).await?;
333    ///
334    /// // Request a response as JSON
335    /// let JsonResponse(resp) = client.post_form::<_, JsonResponse<MyResponse>>("url", &form).await?;
336    ///
337    /// // Expect no response body, just success
338    /// client.post_form::<_, ()>("url", &form).await?;
339    /// # Ok(())
340    /// # }
341    /// ```
342    pub async fn post_form<T, R>(
343        &self,
344        url: &str,
345        payload: &T,
346    ) -> anyhow::Result<R, HttpClientError>
347    where
348        T: Serialize,
349        R: ApiResponse,
350    {
351        let builder = self.client.post(url).form(payload);
352        let response = self.execute_request(builder).await?;
353        R::from_response(response).await
354    }
