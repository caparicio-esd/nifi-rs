1//! # Authentication Module
2//!
3//! Provides bindings for the NiFi "authentication" API endpoints.
4//!
5//! This module is primarily used to query the server's authentication
6//! configuration, such as whether login is supported or if an external
7//! login flow is required.
8
9use crate::common::client::HttpClient;
10use crate::common::config::Config;
11use serde::Deserialize;
12use std::sync::Arc;
13
14/// A service for interacting with NiFi's authentication configuration endpoints.
15///
16/// It is instantiated with shared (`Arc`) instances of `HttpClient` and `Config`.
17pub struct Authentication {
18    client: Arc<HttpClient>,
19    config: Arc<Config>,
20}
21
22/// A top-level wrapper entity for the authentication configuration response.
23///
24/// This struct directly maps to the JSON response from the
25/// `/authentication/configuration` endpoint.
26#[derive(Debug, Deserialize)]
27#[serde(rename_all = "camelCase")]
28pub struct AuthenticationConfigurationEntity {
29    /// The nested configuration DTO.
30    pub authentication_configuration: Option<AuthenticationConfigurationDTO>,
31}
32
33/// Data Transfer Object (DTO) containing the actual authentication configuration details.
34#[derive(Debug, Deserialize)]
35#[serde(rename_all = "camelCase")]
36pub struct AuthenticationConfigurationDTO {
37    /// Whether an external login flow is required.
38    pub external_login_required: Option<bool>,
39    /// Whether this NiFi instance supports login.
40    pub login_supported: Option<bool>,
41    /// The URI to redirect to for login, if external login is used.
42    pub login_uri: Option<String>,
43    /// The URI to redirect to for logout.
44    pub logout_uri: Option<String>,
45}
46
47impl Authentication {
48    /// Creates a new instance of the `Authentication` service.
49    ///
50    /// # Arguments
51    ///
52    /// * `client` - The shared `HttpClient` to be used for requests.
53    /// * `config` - The application configuration (containing `api_base_url`).
54    pub fn new(client: Arc<HttpClient>, config: Arc<Config>) -> Self {
55        Self { client, config }
56    }
57
58    /// Fetches the authentication configuration from the NiFi instance.
59    ///
60    /// This method calls the `GET /authentication/configuration` endpoint.
61    ///
62    /// # Errors
63    ///
64    /// Returns `HttpClientError` if the request fails (e.g., network error,
65    /// HTTP status error, or a JSON parsing error).
66    pub async fn get_authentication_configuration(
67        &self,
68    ) -> anyhow::Result<AuthenticationConfigurationEntity> {
69        let response = self
70            .client
71            .get_json::<AuthenticationConfigurationEntity>(&format!(
72                "{}/authentication/configuration",
73                self.config.api_base_url
74            ))
75            .await?;
76        Ok(response)
77    }
78}

80#[cfg(test)]
81mod test {
82    use super::*;
83    use crate::access::Access;
84    use tracing::debug;
85    use tracing_test::traced_test;
86
87    #[tokio::test]
88    #[traced_test]
89    async fn test_get_authentication_configuration() {
90        // --- 1. Setup ---
91        let client = Arc::new(HttpClient::new());
92        let config = Arc::new(Config::default()); // Assumes correct credentials
93        let access = Access::new(client.clone(), config.clone());
94        let _ = access.get_access_token().await;
95
96        // --- 2. Check Configuration Connection ---
97        let authentication = Authentication::new(client.clone(), config.clone());
98        let auth_configuration = authentication.get_authentication_configuration().await;
99        assert!(
100            auth_configuration.is_ok(),
101            "test_get_authentication_configuration call error: {:?}",
102            auth_configuration
103        );
104
105        // --- 3. Assert over object ---
106        let auth_configuration = auth_configuration.unwrap();
107        assert!(
108            auth_configuration.authentication_configuration.is_some(),
109        );
110        debug!("{:#?}", auth_configuration);
111    }
